/etc/puppet/site/profile/manifests/pip3.pp
	# Install python3-pip and required packages
	class profile::mapleta::pip3 (
		Array $prereqs = [],
		Array $packages = []
	) {
	
	class{'python':
			version    => 'system',
			pip        => 'present',
			dev        => 'present',
			virtualenv => 'absent',
	}

	ensure_packages($prereqs, {
		ensure   => present,
	})

	ensure_packages($packages, {
		ensure   => present,
		provider => 'pip3',
	})
	}
